#### Currency Announcements
```javascript
// Russian currency screen reader formatting
const formatCurrencyForScreenReader = (amount) => {
  const rubles = Math.floor(amount);
  const kopecks = Math.round((amount - rubles) * 100);
  
  let announcement = '';
  
  // Format rubles with Russian grammar
  if (rubles === 0) {
    announcement = '–Ω–æ–ª—å —Ä—É–±–ª–µ–π';
  } else if (rubles === 1) {
    announcement = '–æ–¥–∏–Ω —Ä—É–±–ª—å';
  } else if (rubles >= 2 && rubles <= 4) {
    announcement = `${numberToWords(rubles)} —Ä—É–±–ª—è`;
  } else {
    announcement = `${numberToWords(rubles)} —Ä—É–±–ª–µ–π`;
  }
  
  // Add kopecks if present
  if (kopecks > 0) {
    if (kopecks === 1) {
      announcement += ' –æ–¥–Ω–∞ –∫–æ–ø–µ–π–∫–∞';
    } else if (kopecks >= 2 && kopecks <= 4) {
      announcement += ` ${numberToWords(kopecks)} –∫–æ–ø–µ–π–∫–∏`;
    } else {
      announcement += ` ${numberToWords(kopecks)} –∫–æ–ø–µ–µ–∫`;

  }
  
  return announcement;
};
```
